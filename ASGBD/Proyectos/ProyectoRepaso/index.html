Creacion de tablas


     create table canales(
    id_canal number(2) constraint canales_pk primary key ,
    nombre  varchar(5)
);
create table programas(
    id_programa number(4) constraint programas_pk primary key ,
    nombre varchar2(80) ,
    tiene_capitulos char(2) constraint programas_nn not null
);

create table capitulos(
    n_capitulo number(4),
    id_programa number(4) constraint capitulos_fk references programas(id_programa),
    duracion interval day to second,
    constraint capitulos_pk primary key (n_capitulo,id_programa)
);

create table emisiones(
    n_capitulo number(3),
    id_programa number(4) constraint emisiones_fk1 references programas(id_programa),
    id_canal number(2) constraint emisiones_fk2 references canales(id_canal),
    edad varchar2(200),
    descripcion varchar2(300),
    fecha timestamp,
    constraint emisiones_pk primary key (n_capitulo,id_programa,id_canal,fecha)
                      );
drop table emisiones cascade constraints ;

create view ivan_vista1 as
select distinct c.nombre,p.nombre nombre_programa,e.n_capitulo,e.fecha Fecha_inicio,e.fecha + duracion Fecha_fin from canales c
join emisiones e on c.id_canal = e.id_canal
join programas p on p.id_programa = e.id_programa
join capitulos c2 on e.id_programa = c2.id_programa and e.n_capitulo= c2.n_capitulo
order by c.nombre,fecha;


create view ivan_vista2 as
select distinct p.nombre,c2.n_capitulo,e.edad,e.descripcion,to_char(e.fecha,'dd/mm/yyyy hh24:mi') fecha_inicio, to_char(e.fecha + duracion,'dd/mm/yyyy hh24:mi') fecha_fin from emisiones e
join programas p on p.id_programa = e.id_programa
join capitulos c2 on e.id_programa = c2.id_programa and e.n_capitulo= c2.n_capitulo
where id_canal=1;

create view ivan_vista3 as
select p.nombre,c2.n_capitulo,to_char(e.fecha,'dd/mm/yyyy hh24:mi') fecha from emisiones e
join programas p on e.id_programa = p.id_programa
join capitulos c2 on e.id_programa = c2.id_programa and e.n_capitulo= c2.n_capitulo
where p.nombre= 'MEJOR CONTIGO';

create view ivan_vista5 as
select  distinct p.nombre , c.n_capitulo,e.descripcion from programas p
join capitulos c on p.id_programa = c.id_programa
join emisiones e on p.id_programa = e.id_programa
where descripcion='texto que quieras';


create view ivan_vista6 as
select distinct c.nombre nombre_canal,p.nombre nombre_programa from emisiones e
join programas p on p.id_programa = e.id_programa
join canales c on c.id_canal = e.id_canal
where fecha='fecha que quieras';
